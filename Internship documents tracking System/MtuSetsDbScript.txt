CREATE PROCEDURE [dbo].[SP_StudentLogin]
    @Email NVARCHAR(100),
    @Password NVARCHAR(64)
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @StudentId INT;

    -- Kimlik bilgilerini kontrol et ve TeacherId'yi al
    SELECT @StudentId = Id
    FROM Students 
    WHERE Email = @Email AND Password = @Password;

    -- Eğer TeacherId varsa, giriş kaydı oluştur
    IF @StudentId IS NOT NULL
    BEGIN
        -- Giriş zamanı kayıt et
        INSERT INTO StudentsLoginLogs (Studentıd, Time)
        VALUES (@StudentId, GETDATE());

        -- Sadece gerekli öğretmen bilgilerini döndür
        SELECT *
        FROM Students 
        WHERE Id = @StudentId;
    END
END;
GO

create PROCEDURE [dbo].[SP_UpdateStudent]
    @Id INT,
    @Name nvarchar(50),
	@Email nvarchar(100),
	@Password nvarchar(64),
	@Phone nvarchar(11),
	@SNo nvarchar(11),
	@ImagePath nvarchar(250)
AS
BEGIN
    SET NOCOUNT ON;
    
    -- Eğer yeni isim başka bir kullanıcı tarafından kullanılıyorsa, güncelleme yapma
    IF EXISTS (SELECT 1 FROM Students where (Name = @Name or Email = @Email or Phone = @Phone or SNo = @SNo) AND Id <> @Id)
    BEGIN
        ---- Burada işlem yapılmıyor, çünkü isim başka biri tarafından kullanılıyor
        --RAISERROR('This name is used.', 16, 1);
        RETURN;
    END
    
    -- Eğer isim kullanılmıyorsa, güncellemeyi yap
    UPDATE Students
    SET Name = @Name,Email = @Email, Password = @Password,Phone = @Phone,SNo=@SNo,ImagePath = @ImagePath
    WHERE Id = @Id;
END;
GO
